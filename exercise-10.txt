create database Hotel;

create table Rooms (id int auto_increment primary key, visitor_id int, room_size int, rentable tinyint, with_kitchen tinyint);
create table Floors (id int auto_increment primary key, nb_windows int, with_carpet tinyint);
create table Visitor(id int auto_increment Primary key);
alter table Visitor add column start_date datetime;
alter table Visitor add column end_date datetime;
